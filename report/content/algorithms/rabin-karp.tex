\subsection{The Rabin-Karp Algorithm \cite{cormen2022introduction}}
\subsubsection{Introduction \& Application}
\begin{itemize}
    \item \textbf{Core idea}: The Rabin-Karp algorithm uses a rolling hash function to find occurrences of a pattern $P$ within a text $T$.
    Instead of comparing the pattern character by character for all possible shifts, it calculates a hash value for the pattern and a hash value for each window of the text.
    If the hash values match, it performs a character-by-character comparison to verify if it is a true match or a hash collision.
    The key advantage is that the hash value of the next window can be computed in $\mathcal{O}(1)$ time by dropping the first character and appending the new character.
    \item \textbf{Real-world applications}:
    \begin{itemize}
        \item Plagiarism Detection: Comparing documents to find exact matches of sentences or paragraphs in a large database of academic papers.
        \item Bioinformatics: Searching for specific DNA sequences within massive genetic codes efficiently.
    \end{itemize}
\end{itemize}

\subsubsection{Pseudocode}
\textbf{Notation:}
\begin{itemize}
    \item \textbf{Text $T$ ($R$ rows, $C$ columns)}: The 2D grid of characters.
    \item \textbf{Pattern $P$ ($K$ keywords)}: The list of patterns to search for.
    \item \textbf{Modulo $q$}: A large prime number used to prevent integer overflow and minimize hash collisions (e.g., $q = 1000000007$).
    \item \textbf{Base $d$}: The number of characters in the alphabet (e.g., $d = 256$ for ASCII).
\end{itemize}

\textsc{Rabin-Karp-2D-Matcher}($T, P, d, q$)
\begin{algorithmic}[1]
\State $R = \mathrm{rows}[T]$, $C = \mathrm{cols}[T]$, $K = \mathrm{count}[P]$
\For{$p\_idx = 0$ \textbf{to} $K - 1$}
    \State $pat = P[p\_idx]$
    \State $M = \mathrm{length}[pat]$
    \State $h = d^{M-1} \bmod q$
    \State $p\_hash = \text{hash of } pat$
    
    \State \Comment{\textbf{Horizontal Search}}
    \If{$M \le C$}  
        \For{$r = 0$ \textbf{to} $R - 1$}
            \State $t\_hash = \text{hash of } T[r][0..M-1]$
            \For{$c = 0$ \textbf{to} $C - M$}
                \If{$p\_hash == t\_hash$ \textbf{and} $pat == T[r][c..c+M-1]$}
                    \State append $(r, c)$ to $Positions$
                \EndIf
                \If{$c < C - M$}
                    \State $t\_hash = (d \cdot (t\_hash - T[r][c] \cdot h) + T[r][c+M]) \bmod q$
                    \If{$t\_hash < 0$}
                        \State $t\_hash = t\_hash + q$
                    \EndIf
                \EndIf
            \EndFor
        \EndFor
    \EndIf
    
    \State \Comment{\textbf{Vertical Search}}
    \If{$M \le R$} 
        \For{$c = 0$ \textbf{to} $C - 1$}
            \State $t\_hash = \text{hash of } T[0..M-1][c]$
            \For{$r = 0$ \textbf{to} $R - M$}
                \If{$p\_hash == t\_hash$ \textbf{and} $pat == T[r..r+M-1][c]$}
                    \State append $(r, c)$ to $Positions$
                \EndIf
                \If{$r < R - M$}
                    \State $t\_hash = (d \cdot (t\_hash - T[r][c] \cdot h) + T[r+M][c]) \bmod q$
                    \If{$t\_hash < 0$}
                        \State $t\_hash = t\_hash + q$
                    \EndIf
                \EndIf
            \EndFor
        \EndFor
    \EndIf
\EndFor
\end{algorithmic}

\begin{itemize}
    \item \textbf{Text $T$ (length $n$)}: The main text being searched.
    \item \textbf{Pattern $P$ (length $m$)}: The string that we are looking for.
    \item \textbf{Shift ($s$)}: The current offset of the text $T$.
    \item \textbf{Positions ($[s_1, s_2, \dots]$)}: A list storing the indices where a complete match is found.

\textbf{Example:} text $T = [a, a, b, c, a, b, b]$ and pattern $P = [a, b, c]$.
\\
For simplicity, we map characters to integers ($a=1, b=2, c=3$), set base $d=10$, and prime $q=13$.
\\

\subsubsection{Complexity}

\hline
\textbf{Metric} & \textbf{Complexity} & \textbf{Reasoning} \\
\hline
\textbf{Time (Worst)} & $\mathcal{O}(K \cdot m \cdot R \cdot C)$ & Occurs when the hash function generates frequent collisions (e.g., text and patterns share many identical substrings or all 'A's), forcing a full character comparison at every shift. \\
\hline
\textbf{Time (Average)} & $\mathcal{O}(K \cdot R \cdot C)$ & In practice, with a strong hash function and large prime $q$, collisions are rare. The hash updates in $\mathcal{O}(1)$ time, skipping most character comparisons. \\
\hline
\textbf{Time (Best)} & $\mathcal{O}(K \cdot R \cdot C)$ & The hash values never match (except on actual occurrences). The algorithm glides through the text in constant time per shift. \\
\hline
\textbf{Auxiliary Space} & $\mathcal{O}(1)$ & The algorithm uses only a constant amount of extra memory for storing hash values and loop indices (excluding the space needed to store the output results). \\

\subsubsection{Discussion}
\begin{itemize}
    \item In \textbf{Scenario 1} (Impact of Grid Size), where the number of patterns $K$ is small and the grid scales from $10 \times 10$ to $500 \times 500$, the Rabin-Karp algorithm is highly efficient. Due to its $\mathcal{O}(1)$ rolling hash update, the running time behaves almost linearly, similar to $\mathcal{O}(R \cdot C)$.
    \item In \textbf{Scenario 2} (Impact of Dictionary Size), where the grid is fixed at $100 \times 100$ and $K$ increases, our specific implementation loops through the grid separately for each pattern. Therefore, the execution time scales linearly with $K$. While this is faster than Brute-Force in terms of raw character comparisons, it does not fully exploit Rabin-Karp's multi-pattern matching potential (which could theoretically search all patterns simultaneously using a unified hash table).
    \item \textbf{In conclusion}, the Rabin-Karp algorithm provides a significant speedup over Brute-Force by using mathematical hashing to filter out unnecessary comparisons. It is an excellent choice when dealing with large texts, provided a robust prime number $q$ is chosen to avoid worst-case hash collision degradation.

    \item \textbf{Further Reading:} For a more rigorous mathematical proof regarding the probability of false positive hash collisions, please refer to the primary literature cited in Section \ref{sec:reference}.
\end{itemize}
